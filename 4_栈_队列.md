### 1.1. [232] 用栈实现队列

```python
class MyQueue(object):

    def __init__(self):
        self.data = []
        self.help = []


    def push(self, x):
        """
        :type x: int
        :rtype: None
        """
        self.data.append(x)


    def pop(self):
        """
        :rtype: int
        """
        if self.empty(): # 空
            return -1

        if self.help: # help存在，直接取栈顶元素
            return self.help.pop(-1)

        # help不存在
        #  1.数据迁移 data-->help
        while self.data:
            self.help.append( self.data.pop(-1) )
        # 2. help此时存在，直接取栈顶元素
        return self.help.pop(-1)


    def peek(self):
        """
        :rtype: int
        """
        if self.empty(): # 空
            return -1

        if self.help: # help存在，直接取栈顶元素
            return self.help[-1]

        # help不存在
        #  1.数据迁移 data-->help
        while self.data:
            self.help.append( self.data.pop(-1) )
        # 2. help此时存在，直接取栈顶元素
        return self.help[-1]


    def empty(self):
        """
        :rtype: bool
        """
        return len(self.data)==0 and len(self.help)==0
```

### 1.2. [225] 用队列实现栈

class MyStack(object):

    def __init__(self):
        self.data = []
        self.help = []

    def push(self, x):
        """
        :type x: int
        :rtype: None
        """
        self.data.append(x)

    def pop(self):
        """
        :rtype: int
        """
        if self.empty():
            return -1

        while len(self.data) > 1: # 将前n-1个数据，全部搬迁，最后一个数据先不搬迁: data --> help 
            self.help.append( self.data.pop(0) )

		ans = self.data.pop(0) # 最后一个数据就是要弹出的结果

		# 交换data和help
        tmp = self.data
        self.data = self.help
        self.help = tmp

        return ans

    def top(self):
        """
        :rtype: int
        """
        if self.empty():
            return -1

        while len(self.data) > 1: # data --> help
            self.help.append( self.data.pop(0) )
        ans = self.data[0]

        self.help.append( self.data.pop(0) )

        tmp = self.data
        self.data = self.help
        self.help = tmp

        return ans

    def empty(self):
        """
        :rtype: bool
        """
        return not self.data and not self.help
```

class Solution(object):
    def simplifyPath(self, path):
        """
        :type path: str
        :rtype: str
        """
        paths = path.split("/")

        stack = []
        for ch in paths:
            if ch == "" or ch == ".":
                continue
            if ch == "..":
                if stack:
                    stack.pop(-1)
            else:
                stack.append(ch)

        ans = '/' + '/'.join(stack)

        return ans
```

### 2.2. 🔥 [20] 有效的括号  

```python
class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        map = {
            "]" : "[",
            ")" : "(",
            "}" : "{",
        }
        stack = []

        for ch in s:
            if ch in map.values(): # 左括弧，进栈
                stack.append(ch)
            else: # 右括弧
                if not stack: # 栈空了，没有配对的了 
                    return False
                else: # 栈顶 和 ch 不配对
                    if stack.pop(-1) != map[ch]:
                        return False

        if stack: # 还有多出来的左括弧
            return False

        return True
```

### 2.3. [844] 比较含退格的字符串

```
输入：S = "ab#c", T = "ad#c"
输出：true
解释：S 和 T 都会变成 “ac”。
```

```python
class Solution(object):
    def backspaceCompare(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: bool
        """
        def func(s):
            stack = []
            for ch in s:
                if ch == "#":
                    if stack:
                        stack.pop(-1)
                else:
                    stack.append(ch)
            return "".join(stack)

        return func(s) == func(t)
```

class Solution(object):
    def makeGood(self, s):
        """
        :type s: str
        :rtype: str
        """
        stack = []
        for ch in s:
            if stack: # 栈中有元素

### 2.5. [1047] 删除字符串中的所有相邻重复项


输入："abbaca"

输出："ca"

解释：在 "abbaca" 中，我们可以删除 "bb" 由于两字母相邻且相同，这是此时唯一可以执行删除操作的重复项。之后我们得到字符串 "aaca"，其中又只有 "aa" 可以执行重复项删除操作，所以最后的字符串为 "ca"。


```python
class Solution(object):
    def removeDuplicates(self, s):
        """
        :type s: str
        :rtype: str
        """
        stack = []
        for ch in s:
            if stack:
                if stack[-1] == ch:
                    stack.pop(-1)
                    continue
            stack.append(ch)
        return "".join(stack)
```

### 2.6. [1209] 删除字符串中的所有相邻重复项 II


输入：s = "deeedbbcccbdaa", k = 3

输出："aa"
解释： 
1. 先删除 "eee" 和 "ccc"，得到 "ddbbbdaa"
2. 再删除 "bbb"，得到 "dddaa"
3. 最后删除 "ddd"，得到 "aa"


- 解题关键: **栈中存放的是 [ch, ch出现次数]**

```python
class Solution(object):
    def removeDuplicates(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: str
        """
        if len(s) <= k:
            return s

        stack = [] # [ch, ch数量]

        for ch in s:
            if stack: # 栈上有数据
                ch1,num1 = stack[-1] # 取栈顶元素
                if ch1 == ch: # 栈顶元素 == ch
                    if num1+1==k: # 满k个，出栈
                        stack.pop(-1)
                        continue
                    else: # 不满k个，栈顶数量+1
                        stack[-1][1] = stack[-1][1] + 1
                        continue

            stack.append([ch,1])

        ans = ""
        for (ch,num) in stack:
            ans += (ch * num)
        return ans
```

### 2.7. 😄 [394] 字符串解码

给定一个经过编码的字符串，返回它解码后的字符串。

编码规则为: k[encoded_string]，表示其中方括号内部的 encoded_string 正好重复 k 次。注意 k 保证为正整数。

- 每次入栈和出栈都更新`cur_num`和`ans_str`


```c
示例 1：
输入：s = "3[a]2[bc]"
输出："aaabcbc"
示例 2：
输入：s = "3[a2[c]]"
输出："accaccacc"
```

```python
# 解题思路
# 1.返回值: cur_str
# 2.背住更新公式: cur_str = stack_str.pop(-1) + (stack_num.pop(-1) * cur_str)
# 3.ch分为4种情况: 数字、char、[、]
class Solution(object):
    def decodeString(self, s):
        """
        :type s: str
        :rtype: str
        """
        stack_num = [] # 数字
        stack_str = [] # 字符串
        cur_num = 0
        cur_str = ""

        for ch in s:
            if ch>='0' and ch <='9': #数字
                cur_num = cur_num*10 + int(ch)
            elif ch == '[':
                stack_num.append(cur_num)
                stack_str.append(cur_str)
                cur_num = 0
                cur_str = ""
            elif ch == ']':
                cur_str = stack_str.pop(-1) + (stack_num.pop(-1) * cur_str)
            else: # 字符串
                cur_str += ch

        return cur_str
```

### 3.2.1. 🔥[739] 每日温度

```python
# 给定一个整数数组 temperatures ，表示每天的温度，返回一个数组 answer ，其中 answer[i] 是指对于第 i 天，下一个更高温度出现
# 在几天后。如果气温在这之后都不会升高，请在该位置用 0 来代替。 
```

- 栈中存放的元素，是下标，不是nums[i]

```python
class Solution(object):
    def dailyTemperatures(self, temperatures):
        """
        :type temperatures: List[int]
        :rtype: List[int]
        """
        ans = [0] * len(temperatures)

        stack = [] # 单调栈: 存放下标index

        for i in range(len(temperatures)):
            while stack and temperatures[i] > temperatures[stack[-1]]: # 当前元素 > 栈顶元素
                ans[stack[-1]] = i-stack[-1] # 更新结果
                stack.pop(-1)                # 栈顶元素出栈
            stack.append(i)

        return ans
```

### 3.2.2. [402] 移掉K位数字

给定一个以字符串表示的非负整数 *num*，移除这个数中的 *k* 位数字，使得剩下的数字最小。

```
输入: num = "1432219", k = 3
输出: "1219"
解释: 移除掉三个数字 4, 3, 和 2 形成一个新的最小的数字 1219	。
```

```python
class Solution(object):
    def removeKdigits(self, num, k):
        """
        :type num: str
        :type k: int
        :rtype: str
        """
        stack = [] # 栈中保留小的数据

        for ch in num:
            while stack and ch < stack[-1] and k>0: # 单调栈
                stack.pop(-1)
                k -= 1
            stack.append(ch) # 新元素一定会进栈

        while k>0:
            stack.pop(-1)
            k-=1

        ans = ''.join(stack)

        ans = ans.lstrip('0')
        if ans == "":
            return "0"
        else:
            return ans
```



### 3.2.3. [496] 下一个更大元素 I

给定两个 `没有重复元素` 的数组 nums1 和 nums2 ，其中nums1 是 nums2 的子集。找到 nums1 中每个元素在 nums2 中的下一个比其大的值。

nums1 中数字 x 的下一个更大元素是指 x 在 nums2 中对应位置的右边的第一个比 x 大的元素。如果不存在，对应位置输出 -1 。

```
输入: nums1 = [4,1,2], nums2 = [1,3,4,2].
输出: [-1,3,-1]
解释:
    对于num1中的数字4，你无法在第二个数组中找到下一个更大的数字，因此输出 -1。
    对于num1中的数字1，第二个数组中数字1右边的下一个较大数字是 3。
    对于num1中的数字2，第二个数组中没有下一个更大的数字，因此输出 -1。
```

解题思路

1. nums2创建hash
2. 遍历nums1，先找到nums1在nums2中的索引，搜寻索引后面的数据是否比nums1大

```python
class Solution(object):
    def nextGreaterElement(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: List[int]
        """
        ans = [-1] * len(nums1)

        # 将nums2创建一个hash
        map_nums2 = {} # 值nums2[i], 下标i
        for i in range(len(nums2)):
            map_nums2[nums2[i]] = i

        # 遍历nums1每个元素: 找当前节点nums1[i]的下一个最大的数字
        for i in range(len(nums1)): 
            idx_nums2 = map_nums2[nums1[i]] # nums1[i] --> 找到索引
            for j in range(idx_nums2+1, len(nums2)):
                if nums1[i] < nums2[j]:
                    ans[i] = nums2[j]
                    break

        return ans
```
